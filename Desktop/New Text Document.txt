#include <iostream>
#include <algorithm>

using namespace std;
int a[10000],n,k;

int main()
{
    cin >> n >> k;
    int index = 0;
    for (int i = 0; i < k; ++i)
    {
        for (int j = 0; j < n; ++j)
        {
            cin >> a[j];
        }
        for (int j = n-1; j > 0; --j)
        {
            if (a[j-1] < a[j])
            {
                index = j-1;
                j = 0;
            }
        }
        int lowest = index;
        for (int j = index+1;j < n; ++j)
        {
            if (a[j] < a[lowest])
                lowest = j;
        }
        swap(a[index],a[lowest]);
        sort(a+index+1, a+n);
        for (int j = 0; j < n; ++j)
        {
            cout << a[j];
        }
    }
  return 0;
}
